    31→/**
    32→ * Exam taking screen — full flow:
    33→ * 1. Loading skeleton
    34→ * 2. Question-by-question with audio (loops N times), image, options
    35→ * 3. Click sound on every interaction
    36→ * 4. Wrong → shows correct answer in red
    37→ * 5. Correct → green confirmation
    38→ * 6. Auto-advances to next question
    39→ * 7. Final score + review screen
    40→ * 8. Stats auto-update via /api/student/tests/[id]/submit
    41→ */
    42→@Composable
    43→fun ExamScreen(theme: AppTheme, testId: String, onExit: () -> Unit) {
    44→    val scope = rememberCoroutineScope()
    45→    val sound = rememberSoundManager()
    46→    var test by remember { mutableStateOf<TestDetail?>(null) }
    47→    var loading by remember { mutableStateOf(true) }
    48→    var error by remember { mutableStateOf("") }
    49→    var currentIdx by remember { mutableStateOf(0) }
    50→    val answers = remember { mutableStateMapOf<String, Any>() }
    51→    var submitResult by remember { mutableStateOf<SubmitResponse?>(null) }
    52→    var submitting by remember { mutableStateOf(false) }
    53→    // Per-question feedback (after answering, before moving on)
    54→    var questionFeedback by remember { mutableStateOf<QuestionFeedback?>(null) }
    55→    // Timer
    56→    var timeLeft by remember { mutableStateOf(0) }
    57→
    58→    LaunchedEffect(testId) {
    59→        scope.launch {
    60→            try {
    61→                test = AppState.api.getTestDetail(testId).test
    62→                timeLeft = (test?.durationMin ?: 30) * 60
    63→            } catch (e: retrofit2.HttpException) {
    64→                // HTTP error from server
    65→                error = when (e.code()) {
    66→                    401 -> "Please log in again to take this exam."
    67→                    403 -> "This exam is not active or has ended."
    68→                    404 -> "Exam not found. It may have been removed."
    69→                    else -> "Could not load exam. Please try again."
    70→                }
    71→            } catch (e: java.net.UnknownHostException) {
    72→                error = "No internet connection. Please check your network."
    73→            } catch (e: java.io.IOException) {
    74→                error = "Could not connect to server. Please check your internet."
    75→            } catch (e: Exception) {
    76→                error = "Could not load exam. Please try again."
    77→            }
    78→            loading = false
    79→        }
    80→    }
    81→
    82→    // Countdown timer
    83→    LaunchedEffect(test, submitResult) {
    84→        if (test != null && submitResult == null) {
    85→            while (timeLeft > 0) {
    86→                delay(1000)
    87→                timeLeft--
    88→            }
    89→            if (timeLeft == 0 && test != null && submitResult == null && !submitting) {
    90→                submitting = true
    91→                sound.swoosh()
    92→                try {
    93→                    submitResult = AppState.api.submitTest(test!!.id, SubmitRequest(answers.toMap()))
    94→                    sound.success()
    95→                } catch (e: java.net.UnknownHostException) {
    96→                    error = "No internet connection. Please check your network."
    97→                } catch (e: java.io.IOException) {
    98→                    error = "Could not connect to server. Please check your internet."
    99→                } catch (e: Exception) {
   100→                    error = "Could not submit exam. Please try again."
   101→                }
   102→                submitting = false
   103→            }
   104→        }
   105→    }
   106→
   107→    if (loading) {
   108→        SkeletonListScreen(theme, itemCount = 4)
   109→        return
   110→    }
   111→
   112→    if (error.isNotEmpty()) {
   113→        Column(Modifier.fillMaxSize().padding(20.dp), horizontalAlignment = Alignment.CenterHorizontally, verticalArrangement = Arrangement.Center) {
   114→            Icon(Icons.Default.Error, null, tint = theme.errorRed, modifier = Modifier.size(48.dp))
   115→            Spacer(Modifier.height(12.dp))
   116→            Text(error, color = theme.darkText, fontSize = 14.sp, textAlign = TextAlign.Center)
   117→            Spacer(Modifier.height(16.dp))
   118→            Button(onClick = onExit) { Text("Go back") }
   119→        }
   120→        return
   121→    }
   122→
   123→    val t = test ?: return
   124→    val currentQuestion = t.items.getOrNull(currentIdx)
   125→
   126→    // ─── Result screen ──────────────────────────────────────────────────────
   127→    if (submitResult != null) {
   128→        ExamResultScreen(theme, submitResult!!, onExit, sound)
   129→        return
   130→    }
   131→
   132→    // ─── Question screen ────────────────────────────────────────────────────
   133→    Column(modifier = Modifier.fillMaxSize().background(theme.background)) {
   134→        // Header with timer + progress
   135→        Surface(color = theme.primary, shape = RoundedCornerShape(0.dp)) {
   136→            Column(modifier = Modifier.padding(16.dp)) {
   137→                Row(
   138→                    modifier = Modifier.fillMaxWidth(),
   139→                    horizontalArrangement = Arrangement.SpaceBetween,
   140→                    verticalAlignment = Alignment.CenterVertically
   141→                ) {
   142→                    IconButton(onClick = onExit) {
   143→                        Icon(Icons.Default.Close, null, tint = Color.White)
   144→                    }
   145→                    Column(horizontalAlignment = Alignment.CenterHorizontally) {
   146→                        Text("Question ${currentIdx + 1} of ${t.items.size}", color = Color.White, fontSize = 13.sp, fontWeight = FontWeight.SemiBold)
   147→                        val progress = (currentIdx + 1f) / t.items.size
   148→                        LinearProgressIndicator(
   149→                            progress = { progress },
   150→                            color = Color.White,
   151→                            trackColor = Color.White.copy(alpha = 0.3f),
   152→                            modifier = Modifier.width(120.dp).padding(top = 4.dp)
   153→                        )
   154→                    }
   155→                    // Timer
   156→                    Surface(color = Color.White.copy(alpha = 0.2f), shape = RoundedCornerShape(8.dp)) {
   157→                        Row(
   158→                            modifier = Modifier.padding(horizontal = 10.dp, vertical = 6.dp),
   159→                            verticalAlignment = Alignment.CenterVertically
   160→                        ) {
   161→                            Icon(Icons.Default.Timer, null, tint = Color.White, modifier = Modifier.size(14.dp))
   162→                            Spacer(Modifier.width(4.dp))
   163→                            val mm = timeLeft / 60
   164→                            val ss = timeLeft % 60
   165→                            Text(String.format("%02d:%02d", mm, ss), color = Color.White, fontSize = 13.sp, fontWeight = FontWeight.Bold)
   166→                        }
   167→                    }
   168→                }
   169→                Spacer(Modifier.height(8.dp))
   170→                Text(t.title, color = Color.White.copy(alpha = 0.85f), fontSize = 12.sp)
   171→            }
   172→        }
   173→
   174→        // Question content
   175→        if (currentQuestion == null) {
   176→            Box(Modifier.fillMaxSize(), contentAlignment = Alignment.Center) {
   177→                Text("No questions in this test", color = theme.subText)
   178→            }
   179→            return
   180→        }
   181→
   182→        val q = currentQuestion.question
   183→        LazyColumn(
   184→            modifier = Modifier.weight(1f).fillMaxWidth().padding(16.dp),
   185→            verticalArrangement = Arrangement.spacedBy(12.dp)
   186→        ) {
   187→            // Question stem
   188→            item {
   189→                Surface(color = theme.cardBg, shape = RoundedCornerShape(12.dp), modifier = Modifier.fillMaxWidth(), shadowElevation = 1.dp) {
   190→                    Column(modifier = Modifier.padding(16.dp)) {
   191→                        Row(verticalAlignment = Alignment.CenterVertically) {
   192→                            Surface(color = theme.primary, shape = RoundedCornerShape(6.dp), modifier = Modifier.size(28.dp)) {
   193→                                Box(contentAlignment = Alignment.Center, modifier = Modifier.fillMaxSize()) {
   194→                                    Text("${currentIdx + 1}", color = Color.White, fontSize = 13.sp, fontWeight = FontWeight.Bold)
   195→                                }
   196→                            }
   197→                            Spacer(Modifier.width(8.dp))
   198→                            Text(q.difficulty, color = theme.subText, fontSize = 11.sp, fontWeight = FontWeight.SemiBold)
   199→                            Spacer(Modifier.weight(1f))
   200→                            Text("${currentQuestion.points} pts", color = theme.primary, fontSize = 11.sp, fontWeight = FontWeight.Bold)
   201→                        }
   202→                        Spacer(Modifier.height(10.dp))
   203→                        Text(q.stem, color = theme.darkText, fontSize = 15.sp, fontWeight = FontWeight.Medium, modifier = Modifier.verticalScroll(rememberScrollState()))
   204→                    }
   205→                }
   206→            }
   207→
   208→            // Image (if present)
   209→            q.imageUrl?.let { url ->
   210→                item {
   211→                    Surface(color = theme.cardBg, shape = RoundedCornerShape(12.dp), modifier = Modifier.fillMaxWidth(), shadowElevation = 1.dp) {
   212→                        AsyncImage(url = url, modifier = Modifier.fillMaxWidth().heightIn(max = 240.dp).clip(RoundedCornerShape(12.dp)))
   213→                    }
   214→                }
   215→            }
   216→
   217→            // Audio player (if present, with loop)
   218→            q.audioUrl?.let { url ->
   219→                item {
   220→                    AudioPlayerCard(
   221→                        theme = theme,
   222→                        url = url,
   223→                        loopCount = q.audioLoop,
   224→                        loopDelaySec = q.audioLoopDelay,
   225→                        sound = sound
   226→                    )
   227→                }
   228→            }
   229→
   230→            // Answer input
   231→            item {
   232→                AnswerInput(
   233→                    theme = theme,
   234→                    question = q,
   235→                    userAnswer = answers[q.id],
   236→                    feedback = questionFeedback,
   237→                    onAnswer = { ans ->
   238→                        sound.click()
   239→                        answers[q.id] = ans
   240→                    }
   241→                )
   242→            }
   243→        }
   244→
   245→        // Bottom navigation buttons
   246→        Surface(color = theme.white, shadowElevation = 4.dp) {
   247→            Row(
   248→                modifier = Modifier.fillMaxWidth().padding(16.dp),
   249→                horizontalArrangement = Arrangement.SpaceBetween
   250→            ) {
   251→                OutlinedButton(
   252→                    onClick = {
   253→                        sound.click()
   254→                        if (currentIdx > 0) {
   255→                            currentIdx--
   256→                            questionFeedback = null
   257→                        }
   258→                    },
   259→                    enabled = currentIdx > 0,
   260→                    shape = RoundedCornerShape(10.dp)
   261→                ) {
   262→                    Icon(Icons.Default.ArrowBack, null, modifier = Modifier.size(16.dp))
   263→                    Spacer(Modifier.width(4.dp))
   264→                    Text("Prev")
   265→                }
   266→
   267→                if (currentIdx < t.items.size - 1) {
   268→                    Button(
   269→                        onClick = {
   270→                            sound.swoosh()
   271→                            currentIdx++
   272→                            questionFeedback = null
   273→                        },
   274→                        shape = RoundedCornerShape(10.dp),
   275→                        colors = ButtonDefaults.buttonColors(containerColor = theme.primary)
   276→                    ) {
   277→                        Text("Next")
   278→                        Spacer(Modifier.width(4.dp))
   279→                        Icon(Icons.Default.ArrowForward, null, modifier = Modifier.size(16.dp))
   280→                    }
   281→                } else {
   282→                    Button(
   283→                        onClick = {
   284→                            sound.swoosh()
   285→                            submitting = true
   286→                            scope.launch {
   287→                                try {
   288→                                    submitResult = AppState.api.submitTest(t.id, SubmitRequest(answers.toMap()))
   289→                                    sound.success()
   290→                                } catch (e: java.net.UnknownHostException) {
   291→                                    error = "No internet connection. Please check your network."
   292→                                } catch (e: java.io.IOException) {
   293→                                    error = "Could not connect to server. Please check your internet."
   294→                                } catch (e: Exception) {
   295→                                    error = "Could not submit exam. Please try again."
   296→                                }
   297→                                submitting = false
   298→                            }
   299→                        },
   300→                        shape = RoundedCornerShape(10.dp),
   301→                        colors = ButtonDefaults.buttonColors(containerColor = theme.accent),
   302→                        enabled = !submitting
   303→                    ) {
   304→                        if (submitting) {
   305→                            CircularProgressIndicator(color = Color.White, modifier = Modifier.size(16.dp), strokeWidth = 2.dp)
   306→                        } else {
   307→                            Icon(Icons.Default.Check, null, modifier = Modifier.size(16.dp))
   308→                            Spacer(Modifier.width(4.dp))
   309→                            Text("Submit")
   310→                        }
   311→                    }
   312→                }
   313→            }
   314→        }
   315→    }
   316→}

   320→// ─── Audio player with loop support ───────────────────────────────────────────
   321→@Composable
   322→fun AudioPlayerCard(theme: AppTheme, url: String, loopCount: Int, loopDelaySec: Int, sound: SoundManager) {
   323→    val context = LocalContext.current
   324→    var mediaPlayer by remember { mutableStateOf<android.media.MediaPlayer?>(null) }
   325→    var isPlaying by remember { mutableStateOf(false) }
   326→    var playCount by remember { mutableStateOf(0) }
   327→    val totalPlays = if (loopCount == -1) "∞" else (loopCount + 1).toString()
   328→    val scope = rememberCoroutineScope()
   329→
   330→    DisposableEffect(url) {
   331→        onDispose {
   332→            mediaPlayer?.release()
   333→            mediaPlayer = null
   334→        }
   335→    }
   336→
   337→    Surface(color = theme.cardBg, shape = RoundedCornerShape(12.dp), modifier = Modifier.fillMaxWidth(), shadowElevation = 1.dp) {
   338→        Column(modifier = Modifier.padding(14.dp)) {
   339→            Row(verticalAlignment = Alignment.CenterVertically) {
   340→                Surface(color = theme.primary.copy(alpha = 0.15f), shape = RoundedCornerShape(10.dp), modifier = Modifier.size(40.dp)) {
   341→                    Box(contentAlignment = Alignment.Center, modifier = Modifier.fillMaxSize()) {
   342→                        Icon(Icons.Default.Headphones, null, tint = theme.primary, modifier = Modifier.size(20.dp))
   343→                    }
   344→                }
   345→                Spacer(Modifier.width(10.dp))
   346→                Column(modifier = Modifier.weight(1f)) {
   347→                    Text("Audio question", color = theme.darkText, fontSize = 13.sp, fontWeight = FontWeight.SemiBold)
   348→                    Text(
   349→                        if (loopCount == -1) "Loops continuously"
   350→                        else if (loopCount == 0) "Plays once"
   351→                        else "Plays ${loopCount + 1} times • ${loopDelaySec}s delay",
   352→                        color = theme.subText, fontSize = 11.sp
   353→                    )
   354→                }
   355→                IconButton(onClick = {
   356→                    sound.click()
   357→                    if (isPlaying) {
   358→                        mediaPlayer?.pause()
   359→                        isPlaying = false
   360→                    } else {
   361→                        // Play with looping
   362→                        try {
   363→                            mediaPlayer?.release()
   364→                            val mp = android.media.MediaPlayer().apply {
   365→                                setDataSource(url)
   366→                                setOnPreparedListener {
   367→                                    start()
   368→                                    isPlaying = true
   369→                                    playCount = 1
   370→                                }
   371→                                setOnCompletionListener {
   372→                                    if (loopCount == -1 || playCount < loopCount + 1) {
   373→                                        // Schedule next play (with delay if set)
   374→                                        scope.launch {
   375→                                            if (loopDelaySec > 0) delay(loopDelaySec * 1000L)
   376→                                            if (loopCount == -1 || playCount < loopCount + 1) {
   377→                                                playCount++
   378→                                                start()
   379→                                            } else {
   380→                                                isPlaying = false
   381→                                            }
   382→                                        }
   383→                                    } else {
   384→                                        isPlaying = false
   385→                                    }
   386→                                }
   387→                                setOnErrorListener { _, _, _ ->
   388→                                    isPlaying = false
   389→                                    true
   390→                                }
   391→                                prepareAsync()
   392→                            }
   393→                            mediaPlayer = mp
   394→                        } catch (_: Exception) {
   395→                            isPlaying = false
   396→                        }
   397→                    }
   398→                }) {
   399→                    Icon(
   400→                        if (isPlaying) Icons.Default.Pause else Icons.Default.PlayArrow,
   401→                        null,
   402→                        tint = theme.primary,
   403→                        modifier = Modifier.size(28.dp)
   404→                    )
   405→                }
   406→            }
   407→            if (loopCount > 0 || loopCount == -1) {
   408→                Spacer(Modifier.height(6.dp))
   409→                Text("Play $playCount / $totalPlays", color = theme.subText, fontSize = 10.sp)
   410→            }
   411→        }
   412→    }
   413→}

   415→// ─── Answer input (different per question type) ───────────────────────────────
   416→@Composable
   417→fun AnswerInput(
   418→    theme: AppTheme,
   419→    question: QuestionDetail,
   420→    userAnswer: Any?,
   421→    feedback: QuestionFeedback?,
   422→    onAnswer: (Any) -> Unit
   423→) {
   424→    val options = question.options ?: emptyList()
   425→
   426→    Column(verticalArrangement = Arrangement.spacedBy(8.dp)) {
   427→        when {
   428→            question.type == "SINGLE_CHOICE" || question.type == "TRUE_FALSE" -> {
   429→                options.forEachIndexed { i, opt ->
   430→                    val selected = userAnswer == opt
   431→                    val isCorrectFeedback = feedback?.let { feedback.correctAnswer == opt }
   432→                    val isWrongSelected = feedback != null && selected && !feedback.isCorrect
   433→                    val bgColor = when {
   434→                        isCorrectFeedback == true -> Color(0xFFD4EDDA)
   435→                        isWrongSelected -> Color(0xFFFFCDD2)
   436→                        selected -> theme.primary.copy(alpha = 0.1f)
   437→                        else -> theme.cardBg
   438→                    }
   439→                    val borderColor = when {
   440→                        isCorrectFeedback == true -> Color(0xFF28A745)
   441→                        isWrongSelected -> theme.errorRed
   442→                        selected -> theme.primary
   443→                        else -> theme.divider
   444→                    }
   445→                    Surface(
   446→                        color = bgColor,
   447→                        shape = RoundedCornerShape(10.dp),
   448→                        border = androidx.compose.foundation.BorderStroke(1.5.dp, borderColor),
   449→                        modifier = Modifier.fillMaxWidth().clickable {
   450→                            if (feedback == null) onAnswer(opt)
   451→                        }
   452→                    ) {
   453→                        Row(modifier = Modifier.padding(14.dp), verticalAlignment = Alignment.CenterVertically) {
   454→                            Surface(
   455→                                color = if (selected) theme.primary else Color.Transparent,
   456→                                shape = RoundedCornerShape(50),
   457→                                modifier = Modifier.size(20.dp)
   458→                            ) {
   459→                                if (selected) {
   460→                                    Box(contentAlignment = Alignment.Center, modifier = Modifier.fillMaxSize()) {
   461→                                        Icon(Icons.Default.Check, null, tint = Color.White, modifier = Modifier.size(14.dp))
   462→                                    }
   463→                                }
   464→                            }
   465→                            Spacer(Modifier.width(12.dp))
   466→                            Text(opt, color = theme.darkText, fontSize = 14.sp, modifier = Modifier.weight(1f))
   467→                            if (isCorrectFeedback == true) {
   468→                                Icon(Icons.Default.CheckCircle, null, tint = Color(0xFF28A745))
   469→                            } else if (isWrongSelected) {
   470→                                Icon(Icons.Default.Cancel, null, tint = theme.errorRed)
   471→                            }
   472→                        }
   473→                    }
   474→                }
   475→            }
   476→            question.type == "MULTIPLE_CHOICE" -> {
   477→                val selectedList = (userAnswer as? List<*>)?.filterIsInstance<String>() ?: emptyList()
   478→                options.forEachIndexed { i, opt ->
   479→                    val selected = selectedList.contains(opt)
   480→                    Surface(
   481→                        color = if (selected) theme.primary.copy(alpha = 0.1f) else theme.cardBg,
   482→                        shape = RoundedCornerShape(10.dp),
   483→                        border = androidx.compose.foundation.BorderStroke(1.5.dp, if (selected) theme.primary else theme.divider),
   484→                        modifier = Modifier.fillMaxWidth().clickable {
   485→                            val newList = if (selected) selectedList - opt else selectedList + opt
   486→                            onAnswer(newList)
   487→                        }
   488→                    ) {
   489→                        Row(modifier = Modifier.padding(14.dp), verticalAlignment = Alignment.CenterVertically) {
   490→                            Surface(
   491→                                color = if (selected) theme.primary else Color.Transparent,
   492→                                shape = RoundedCornerShape(4.dp),
   493→                                modifier = Modifier.size(20.dp)
   494→                            ) {
   495→                                if (selected) {
   496→                                    Box(contentAlignment = Alignment.Center, modifier = Modifier.fillMaxSize()) {
   497→                                        Icon(Icons.Default.Check, null, tint = Color.White, modifier = Modifier.size(14.dp))
   498→                                    }
   499→                                }
   500→                            }
   501→                            Spacer(Modifier.width(12.dp))
   502→                            Text(opt, color = theme.darkText, fontSize = 14.sp, modifier = Modifier.weight(1f))
   503→                        }
   504→                    }
   505→                }
   506→            }
   507→            else -> {
   508→                // ONE_WORD / SHORT_ANSWER / FILL_BLANK
   509→                OutlinedTextField(
   510→                    value = (userAnswer as? String) ?: "",
   511→                    onValueChange = { onAnswer(it) },
   512→                    label = { Text("Your answer") },
   513→                    modifier = Modifier.fillMaxWidth(),
   514→                    colors = OutlinedTextFieldDefaults.colors(
   515→                        focusedTextColor = theme.darkText,
   516→                        unfocusedTextColor = theme.darkText,
   517→                        focusedBorderColor = theme.primary,
   518→                        unfocusedBorderColor = theme.divider,
   519→                        cursorColor = theme.primary
   520→                    ),
   521→                    shape = RoundedCornerShape(10.dp),
   522→                    singleLine = true
   523→                )
   524→            }
   525→        }
   526→    }
   527→}

   546→// ─── Result screen ────────────────────────────────────────────────────────────
   547→@Composable
   548→fun ExamResultScreen(theme: AppTheme, result: SubmitResponse, onExit: () -> Unit, sound: SoundManager) {
   549→    LaunchedEffect(Unit) { sound.success() }
   550→
   551→    LazyColumn(
   552→        modifier = Modifier.fillMaxSize().background(theme.background).padding(16.dp),
   553→        verticalArrangement = Arrangement.spacedBy(12.dp)
   554→    ) {
   555→        // Score card
   556→        item {
   557→            Surface(
   558→                color = theme.cardBg,
   559→                shape = RoundedCornerShape(20.dp),
   560→                modifier = Modifier.fillMaxWidth(),
   561→                shadowElevation = 3.dp
   562→            ) {
   563→                Column(modifier = Modifier.padding(24.dp), horizontalAlignment = Alignment.CenterHorizontally) {
   564→                    val pct = if (result.maxScore > 0) (result.score * 100 / result.maxScore) else 0
   565→                    val passed = pct >= 40
   566→                    Surface(
   567→                        color = if (passed) Color(0xFF4CAF50) else theme.errorRed,
   568→                        shape = RoundedCornerShape(50),
   569→                        modifier = Modifier.size(80.dp)
   570→                    ) {
   571→                        Box(contentAlignment = Alignment.Center, modifier = Modifier.fillMaxSize()) {
   572→                            Icon(
   573→                                if (passed) Icons.Default.Check else Icons.Default.Close,
   574→                                null,
   575→                                tint = Color.White,
   576→                                modifier = Modifier.size(40.dp)
   577→                            )
   578→                        }
   579→                    }
   580→                    Spacer(Modifier.height(16.dp))
   581→                    Text(
   582→                        if (passed) "Congratulations!" else "Keep practicing",
   583→                        color = theme.darkText,
   584→                        fontSize = 20.sp,
   585→                        fontWeight = FontWeight.Bold
   586→                    )
   587→                    Spacer(Modifier.height(8.dp))
   588→                    Text("$pct%", color = if (passed) Color(0xFF4CAF50) else theme.errorRed, fontSize = 40.sp, fontWeight = FontWeight.Bold)
   589→                    Text("${result.score} / ${result.maxScore} points", color = theme.subText, fontSize = 13.sp)
   590→                    Spacer(Modifier.height(16.dp))
   591→                    Button(
   592→                        onClick = { sound.click(); onExit() },
   593→                        modifier = Modifier.fillMaxWidth().height(48.dp),
   594→                        shape = RoundedCornerShape(10.dp),
   595→                        colors = ButtonDefaults.buttonColors(containerColor = theme.primary)
   596→                    ) {
   597→                        Text("Back to tests", fontWeight = FontWeight.SemiBold)
   598→                    }
   599→                }
   600→            }
   601→        }
   602→
   603→        // Per-question review
   604→        item {
   605→            Text("Review", color = theme.darkText, fontSize = 18.sp, fontWeight = FontWeight.Bold)
   606→        }
   607→
   608→        items(result.review) { review ->
   609→            ReviewCard(theme, review)
   610→        }
   611→    }
   612→}

   614→@Composable
   615→fun ReviewCard(theme: AppTheme, review: ReviewItem) {
   616→    Surface(
   617→        color = theme.cardBg,
   618→        shape = RoundedCornerShape(12.dp),
   619→        modifier = Modifier.fillMaxWidth(),
   620→        shadowElevation = 1.dp
   621→    ) {
   622→        Column(modifier = Modifier.padding(14.dp)) {
   623→            Row(verticalAlignment = Alignment.CenterVertically) {
   624→                Icon(
   625→                    if (review.isCorrect) Icons.Default.CheckCircle else Icons.Default.Cancel,
   626→                    null,
   627→                    tint = if (review.isCorrect) Color(0xFF4CAF50) else theme.errorRed,
   628→                    modifier = Modifier.size(20.dp)
   629→                )
   630→                Spacer(Modifier.width(8.dp))
   631→                Text(
   632→                    if (review.isCorrect) "Correct" else "Incorrect",
   633→                    color = if (review.isCorrect) Color(0xFF4CAF50) else theme.errorRed,
   634→                    fontSize = 12.sp,
   635→                    fontWeight = FontWeight.SemiBold
   636→                )
   637→            }
   638→            Spacer(Modifier.height(8.dp))
   639→            Text(review.stem.take(200), color = theme.darkText, fontSize = 13.sp, fontWeight = FontWeight.Medium)
   640→            Spacer(Modifier.height(6.dp))
   641→            review.options?.let { opts ->
   642→                opts.forEachIndexed { i, opt ->
   643→                    val isUserAns = (review.userAnswer == opt) || ((review.userAnswer as? List<*>)?.contains(opt) == true)
   644→                    val isCorrectAns = (review.correctAnswer == opt) || ((review.correctAnswer as? List<*>)?.contains(opt) == true)
   645→                    val bg = when {
   646→                        isCorrectAns -> Color(0xFFD4EDDA)
   647→                        isUserAns && !isCorrectAns -> Color(0xFFFFCDD2)
   648→                        else -> Color.Transparent
   649→                    }
   650→                    Surface(color = bg, shape = RoundedCornerShape(6.dp), modifier = Modifier.fillMaxWidth().padding(vertical = 2.dp)) {
   651→                        Row(modifier = Modifier.padding(8.dp), verticalAlignment = Alignment.CenterVertically) {
   652→                            Text("${'A' + i}.", color = theme.subText, fontSize = 12.sp, fontWeight = FontWeight.Bold)
   653→                            Spacer(Modifier.width(8.dp))
   654→                            Text(opt, color = theme.darkText, fontSize = 12.sp, modifier = Modifier.weight(1f))
   655→                            if (isCorrectAns) Icon(Icons.Default.Check, null, tint = Color(0xFF28A745), modifier = Modifier.size(14.dp))
   656→                        }
   657→                    }
   658→                }
   659→            }
   660→            if (!review.explanation.isNullOrBlank()) {
   661→                Spacer(Modifier.height(8.dp))
   662→                Surface(color = theme.primary.copy(alpha = 0.05f), shape = RoundedCornerShape(6.dp)) {
   663→                    Column(modifier = Modifier.padding(8.dp)) {
   664→                        Text("Explanation", color = theme.primary, fontSize = 10.sp, fontWeight = FontWeight.SemiBold)
   665→                        Text(review.explanation, color = theme.darkText, fontSize = 12.sp)
   666→                    }
   667→                }
   668→            }
   669→        }
   670→    }
   671→}
