    44→// Email/phone validation helpers
    45→private fun isEmail(s: String) = android.util.Patterns.EMAIL_ADDRESS.matcher(s).matches()
    46→private fun isValidPhone(s: String): Boolean {
    47→    val digits = s.replace("\\D".toRegex(), "")
    48→    return digits.length in 7..15
    49→}
